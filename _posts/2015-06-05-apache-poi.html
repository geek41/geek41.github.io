用Apache POI处理Excel

1 安装

<dependency>
  <groupId>org.apache.poi</groupId>
  <artifactId>poi</artifactId>
  <version>3.12</version>
</dependency>

<!-- WorkbookFactory and friends all require poi-ooxml, not just core poi -->
<dependency>
  <groupId>org.apache.poi</groupId>
  <artifactId>poi-ooxml</artifactId>
  <version>3.12</version>
</dependency>
